# 91 - 开发命令速查

> 日常开发常用命令小抄。Windows + WSL/Cmd/PowerShell 通用。

---

## 1. Docker

```bash
# 启动 / 停止
cd docker
docker compose up -d                # 后台启动
docker compose down                 # 停止
docker compose down -v              # 停止 + 删数据卷（破坏性）
docker compose restart mysql        # 重启单个服务

# 状态
docker ps                           # 看运行中容器
docker ps -a                        # 看所有（含已停止）
docker logs moveme-mysql --tail 100
docker logs -f moveme-redis         # 跟踪日志

# 进入容器
docker exec -it moveme-mysql bash
docker exec -it moveme-mysql mysql -uroot -proot123 moveme
docker exec -it moveme-redis redis-cli

# 一次性命令
docker exec moveme-mysql mysqladmin ping -uroot -proot123 --silent
docker exec moveme-mysql mysql -uroot -proot123 -e "SHOW DATABASES"

# 清理
docker system prune                 # 清没用的镜像/容器/网络
docker volume ls                    # 看数据卷
docker volume rm <name>             # 删数据卷
```

## 2. MySQL（容器内）

```bash
# 进入 mysql shell
docker exec -it moveme-mysql mysql -uroot -proot123 moveme

# 一次性 SQL
docker exec moveme-mysql mysql -uroot -proot123 moveme -e "SHOW TABLES"

# 导出数据
docker exec moveme-mysql mysqldump -uroot -proot123 moveme > backup.sql

# 导入数据
docker exec -i moveme-mysql mysql -uroot -proot123 moveme < backup.sql

# 看表行数
docker exec moveme-mysql mysql -uroot -proot123 moveme -e "
  SELECT table_name, table_rows
  FROM information_schema.tables
  WHERE table_schema='moveme'
  ORDER BY table_rows DESC;
"

# 看 schema
docker exec moveme-mysql mysql -uroot -proot123 moveme -e "DESC movies"
docker exec moveme-mysql mysql -uroot -proot123 moveme -e "SHOW INDEX FROM movies"

# 慢查询
docker exec moveme-mysql mysql -uroot -proot123 -e "
  SET GLOBAL slow_query_log='ON';
  SET GLOBAL long_query_time=0.5;
"
```

## 4. Maven (mvnw)

```bash
cd moveme-backend

# Windows
mvnw.cmd <goal>
# Linux/Mac
./mvnw <goal>

# 常用 goal
mvnw.cmd clean                     # 清 target/
mvnw.cmd compile                   # 编译
mvnw.cmd test                      # 跑测试
mvnw.cmd package -DskipTests       # 打 jar
mvnw.cmd spring-boot:run           # 开发模式启动
mvnw.cmd dependency:tree           # 依赖树
mvnw.cmd dependency:tree -Dincludes=org.springframework.*

# 跳测试构建
mvnw.cmd clean package -DskipTests

# 单跑一个测试类
mvnw.cmd test -Dtest=UserServiceTest

# Maven Wrapper 报错时备用
java -Dmaven.multiModuleProjectDirectory=. -cp .mvn/wrapper/maven-wrapper.jar org.apache.maven.wrapper.MavenWrapperMain spring-boot:run
```

```yaml
mybatis-plus:
```

```bash
mvnw.cmd spring-boot:run -Dspring-boot.run.jvmArguments="-agentlib:jdwp=transport=dt_socket,server=y,suspend=n,address=*:5005"

```

## 11. 一行命令做的事

```bash
# 重置数据库 + 启后端 + 启前端（三窗口）
docker compose -f docker/docker-compose.yml down -v && \
  docker compose -f docker/docker-compose.yml up -d

# 等 mysql healthy
while ! docker exec moveme-mysql mysqladmin ping -uroot -proot123 --silent; do sleep 1; done

# 终端 1
cd moveme-backend && ./mvnw spring-boot:run

# 终端 2
cd moveme-frontend && npm run dev
```
